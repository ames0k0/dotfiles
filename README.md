```
vim default Leader -> '\'
let mapleader=','

./install.sh [ args ]
    -tc     -> *make tmux config
    -al:b|z -> *make bash|zsh aliases
    -vi     -> install vim
    -vc     -> *make vim config
    -p2|3   -> install modules for python2|3

    *make   -> cp exist file with config
```
